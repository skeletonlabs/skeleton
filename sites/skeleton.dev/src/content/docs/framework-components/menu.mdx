## Anatomy

Here's an overview of how the Menu component is structured in code:

<Framework id="react">

```tsx
import { Menu, Portal } from '@skeletonlabs/skeleton-react';

export default function Anatomy() {
	return (
		<Menu>
			<Menu.Trigger />
			{/* or <Menu.ContextTrigger /> for right-click */}
			<Portal>
				<Menu.Positioner>
					<Menu.Content>
						{/* Simple items */}
						<Menu.Item value="..." />
						<Menu.Separator />
						
						{/* Grouped items */}
						<Menu.ItemGroup>
							<Menu.ItemGroupLabel />
							<Menu.Item value="..." />
						</Menu.ItemGroup>
						
						{/* Option items (checkbox/radio) */}
						<Menu.OptionItem type="checkbox" name="..." value="..." />
						
						{/* Optional arrow */}
						<Menu.Arrow>
							<Menu.ArrowTip />
						</Menu.Arrow>
					</Menu.Content>
				</Menu.Positioner>
			</Portal>
		</Menu>
	);
}
```

</Framework>
