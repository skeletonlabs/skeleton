## Anatomy

Here's an overview of how the Menu component is structured in code:

<Framework id="react">

```tsx
import { Menu, Portal } from '@skeletonlabs/skeleton-react';

export default function Anatomy() {
	return (
		<Menu>
			<Menu.Trigger />
			<Menu.ContextTrigger />
			<Portal>
				<Menu.Positioner>
					<Menu.Content>
						<Menu.Item />
						<Menu.Separator />
						<Menu.ItemGroup>
							<Menu.ItemGroupLabel />
							<Menu.Item />
							<Menu.OptionItem />
						</Menu.ItemGroup>
						<Menu.Arrow>
							<Menu.ArrowTip />
						</Menu.Arrow>
					</Menu.Content>
				</Menu.Positioner>
			</Portal>
		</Menu>
	);
}
```

</Framework>
